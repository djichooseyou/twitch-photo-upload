const API_BASE = "https://twitch-photo-upload.onrender.com"; form.addEventListener("submit", async (e) => { e.preventDefault(); btn.innerText = "Uploading..."; btn.classList.remove("error","success"); const formData = new FormData(form); try { const res = await fetch(API_BASE + "/upload", { method: "POST", body: formData }); if(res.ok){ btn.innerText = "Uploaded!"; btn.classList.add("success"); form.reset(); } else { btn.innerText = "Error"; btn.classList.add("error"); } } catch(err){ btn.innerText = "Error"; btn.classList.add("error"); } });
